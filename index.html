Classes


class Point:
    def __init__(self, x, y):
        self.x = x
        self.y = y
    def __str__(self):
        return "({x}, {y})".format(x=self.x, y=self.y)
						

class Point {
    constructor(x, y) {
        this.x = x;
        this.y = y;
    }
    toString() {
        return `(${this.x}, ${this.y})`;
    }
}

Sub-Classes


class ColorPoint(Point):
    def __init__(self, x, y, color):
        super().__init__(x, y)
        self.color = color
    def __str__(self):
        return "{super} in color {color}".format(
            super=super().__str__(),
            color=self.color
        )
						

class ColorPoint extends Point {
    constructor(x, y, color) {
        super(x, y);
        this.color = color;
    }
    toString() {
        return `${super.toString()} in ${this.color}`;
    }
}

Getters/Setters


from math import sqrt
class SmartPoint(Point):
    @property
    def hypotenuse(self):
        return sqrt(self.x ** 2 + self.y ** 2)
    @hypotenuse.setter
    def hypotenuse(self, z):
        # Sily setter example
        self.y = sqrt(z ** 2 - self.x ** 2)
						

class SmartPoint extends Point {
    get hypotenuse() {
        return Math.sqrt(this.x ** 2 + this.y ** 2);
    }
    set hypotenuse(z) {
        // Silly setter example
        this.y = Math.sqrt(z ** 2 - this.x ** 2);
    }
}

Getters/Setters


point = SmartPoint(3, 4)
